static void led1_thread_entry(void *parameter){
    while (1)
    {
        LED1_ON;

}

static void led2_thread_entry(void *parameter)
{
    while (1)
    {
        LED2_ON;
        rt_thread_delay(500); /* 延时 500 个 tick */
        LED2_OFF;
        rt_thread_delay(500); /* 延时 500 个 tick */

void SysTick_Handler(void)
{
    /* enter interrupt */
    rt_interrupt_enter();

    rt_tick_increase();

    /* leave interrupt */
    rt_interrupt_leave();
}
